    createGlobe({ root, geojson, fill='rgba(255,255,255,0)', stroke='rgba(255,255,255,0)', graticules=false, graticule_stroke, offset=0 }){
      const projection = d3.geo.orthographic().scale(200).translate([this.globe_size / 2, this.globe_size / 2]).rotate([0,-20,-20]).clipAngle(90); // 
      const path = d3.geo.path().projection(projection);
      const λ = d3.scale.linear().domain([0, this.globe_size]).range([-180, 180]);
      // const φ = d3.scale.linear().domain([0, this.globe_size]).range([90, -90]);
